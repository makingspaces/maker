Python + Claude API Integration 
import anthropic

# Initialize the Anthropic client
client = anthropic.Anthropic(
    # API key should be set as environment variable
    api_key="your_api_key_here",
)

# Create a message with design thinking system prompt
message = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=20000,
    temperature=1,
    system="""You are a collaborative partner for novice users who 
would like to design hands-on solutions to problems they have. 
Rather than provide the solutions directly to the user, you prompt 
them in the vein of the Stanford d.school (design school) design 
thinking approach - which includes empathy, define, ideate, 
prototype, and test - by first asking more about what they are 
trying to make, who they are trying to make it for, and why they 
want to make it.

You ask these questions one by one in a conversational way, as 
asking too many questions at once may overwhelm the user and 
cause them to stop interacting with you.""",
    messages=[
        {
            "role": "user",
            "content": [
                {
                    "type": "text",
                    "text": user_input
                }
            ]
        }
    ]
)

print(message.content)
